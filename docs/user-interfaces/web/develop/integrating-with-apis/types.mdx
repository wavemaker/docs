# Types

Type definitions and TypeScript best practices for API integration.

## Overview

Using TypeScript types ensures type safety when working with APIs, providing better developer experience, catching errors early, and serving as documentation.

## API Response Types

### Basic Response Types

```typescript
// User type
interface User {
  id: string;
  name: string;
  email: string;
  avatar?: string;
  createdAt: string;
  updatedAt: string;
}

interface Product {
  id: string;
  name: string;
  description: string;
  price: number;
  category: string;

// Paginated response

  data: T[];
  pagination: {
    page: number;
    limit: number;
```

## Generic API Types

### Generic Fetch Hook Type

```typescript
interface UseFetchResult<T> {
  data: T | null;
  loading: boolean;
  error: string | null;
  refetch: () => void;
}

function useFetch<T>(url: string): UseFetchResult<T> {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState<boolean>(true);
  const [error, setError] = useState<string | null>(null);

  const fetchData = async () => {
    try {
      setLoading(true);
      const response = await fetch(url);
      const result: T = await response.json();
      setData(result);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Unknown error');
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchData();
  }, [url]);

  return { data, loading, error, refetch: fetchData };
}

// Usage with type inference
const UserProfile = ({ userId }: { userId: string }) => {
  const { data: user, loading, error } = useFetch<User>(`/api/users/${userId}`);

  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;
  return <div>{user?.name}</div>;
};
```

### Generic API Client Type

```typescript
interface ApiClient {
  get<T>(url: string, options?: ApiRequestOptions): Promise<T>;
  post<T>(url: string, data: any, options?: ApiRequestOptions): Promise<T>;
  put<T>(url: string, data: any, options?: ApiRequestOptions): Promise<T>;
  patch<T>(url: string, data: any, options?: ApiRequestOptions): Promise<T>;
  delete<T>(url: string, options?: ApiRequestOptions): Promise<T>;
}

class FetchApiClient implements ApiClient {
  constructor(private baseURL: string) {}

  async get<T>(url: string, options?: ApiRequestOptions): Promise<T> {
    const response = await fetch(`${this.baseURL}${url}`, {
      method: 'GET',
      ...options,
    });
    return response.json();
  }

  async post<T>(url: string, data: any, options?: ApiRequestOptions): Promise<T> {
    const response = await fetch(`${this.baseURL}${url}`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        ...options?.headers,
      },
      body: JSON.stringify(data),
      ...options,
    });
    return response.json();
  }

  // Implement other methods...
}

// Usage
const apiClient = new FetchApiClient('https://api.example.com');
const user = await apiClient.get<User>('/users/123');
const newUser = await apiClient.post<User>('/users', createUserData);
```

## Union and Intersection Types

### Union Types

```typescript

// Discriminated unions

  | { status: 'loading' }

      return <div>Loading...</div>;
    case 'success':
      return <div>{JSON.stringify(state.data)}</div>;
    case 'error':
      return <div>Error: {state.error}</div>;
  }
};
```

### Custom Utility Types

```typescript
// Make specific properties optional
type Optional<T, K extends keyof T> = Omit<T, K> & Partial<Pick<T, K>>;

type UserWithOptionalEmail = Optional<User, 'email'>;

// Make specific properties required
type RequiredKeys<T, K extends keyof T> = Omit<T, K> & Required<Pick<T, K>>;

// API response wrapper type
```

## Type Guards

### Type Checking Functions

```typescript
function isUser(obj: any): obj is User {
  return (
    typeof obj === 'object' &&
    typeof obj.id === 'string' &&
    typeof obj.name === 'string' &&
    typeof obj.email === 'string'
  );
}

function isApiError(response: any): response is ErrorResponse {
  return response.success === false && 'error' in response;
}

function isSuccessResponse<T>(
  response: ApiResponse<T> | ErrorResponse
): response is ApiResponse<T> {
  return response.success === true;
}

// Usage
const handleResponse = (response: ApiResponse<User> | ErrorResponse) => {
  if (isSuccessResponse(response)) {
    console.log('User:', response.data);
  } else if (isApiError(response)) {
    console.error('Error:', response.error.message);
  }
};
```

## Type Inference

### Inferred Return Types

```typescript
// Automatic type inference
const fetchUser = async (id: string) => {
  const response = await fetch(`/api/users/${id}`);
  const user: User = await response.json();
  return user;
};

// Type is inferred as Promise<User>
type FetchUserReturn = ReturnType<typeof fetchUser>;

// Extract promise type
type UnwrapPromise<T> = T extends Promise<infer U> ? U : T;
type UserType = UnwrapPromise<ReturnType<typeof fetchUser>>; // User
```

### Type Inference with Generics

```typescript
function createApiHook<T>(endpoint: string) {
  return () => {
    const [data, setData] = useState<T | null>(null);
    const [loading, setLoading] = useState(true);

    useEffect(() => {
      fetch(endpoint)
        .then(res => res.json())
        .then((data: T) => setData(data))
        .finally(() => setLoading(false));
    }, []);

    return { data, loading };
  };
}

// Type is automatically inferred
const useUsers = createApiHook<User[]>('/api/users');
const usePosts = createApiHook<Post[]>('/api/posts');
```

## Zod for Runtime Validation

### Schema Validation

```typescript
import { z } from 'zod';

// Define schema
const UserSchema = z.object({
  id: z.string().uuid(),
  name: z.string().min(1),
  email: z.string().email(),
  age: z.number().int().positive().optional(),
  role: z.enum(['admin', 'user', 'guest']),
  createdAt: z.string().datetime(),
});

// Infer TypeScript type from schema
type User = z.infer<typeof UserSchema>;

// Validate API response
const validateUser = (data: unknown): User => {
  return UserSchema.parse(data);
};

// Safe parsing
const safeParseUser = (data: unknown) => {
  const result = UserSchema.safeParse(data);
  if (result.success) {
    return result.data;
  } else {
    console.error('Validation errors:', result.error);
    return null;
  }
};
```

## Type-Safe API Client

```typescript
interface ApiEndpoints {
  getUser: {
    params: { id: string };
    response: User;
  };
  getUsers: {
    params: { page?: number; limit?: number };
    response: PaginatedResponse<User>;
  };
  createUser: {
    body: CreateUserRequest;
    response: User;
  };
  updateUser: {
    params: { id: string };
    body: UpdateProductRequest;
    response: User;
  };
}

class TypedApiClient {
  async get<K extends keyof ApiEndpoints>(
    endpoint: K,
    params: ApiEndpoints[K] extends { params: infer P } ? P : never
  ): Promise<ApiEndpoints[K]['response']> {
    // Implementation
    const response = await fetch(`/api/${endpoint}?${new URLSearchParams(params as any)}`);
    return response.json();
  }

  async post<K extends keyof ApiEndpoints>(
    endpoint: K,
    body: ApiEndpoints[K] extends { body: infer B } ? B : never
  ): Promise<ApiEndpoints[K]['response']> {
    // Implementation
    const response = await fetch(`/api/${endpoint}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });
    return response.json();
  }
}

// Usage with full type safety
const client = new TypedApiClient();
const user = await client.get('getUser', { id: '123' });
const users = await client.get('getUsers', { page: 1, limit: 10 });
const newUser = await client.post('createUser', { name: 'John', email: 'john@example.com', password: 'secret' });
```

## Best Practices

### 1. Use Strict Types

```typescript
// ✅ Good - Strict types
interface User {
  id: string;
  name: string;
  email: string;
}

// ❌ Bad - Loose types
interface User {
  id: any;
  name: any;
  email: any;
}
```

### 2. Avoid Type Assertions

```typescript
// ✅ Good - Type guards
if (isUser(data)) {
  console.log(data.name);
}

// ❌ Bad - Type assertion
console.log((data as User).name);
```

### 3. Use Generics for Reusability

```typescript
// ✅ Good - Generic type
function fetchData<T>(url: string): Promise<T> {
  return fetch(url).then(r => r.json());
}

// ❌ Bad - Repeated types
function fetchUser(url: string): Promise<User> {
  return fetch(url).then(r => r.json());
}
function fetchProduct(url: string): Promise<Product> {
  return fetch(url).then(r => r.json());
}
```
